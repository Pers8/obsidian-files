```SQL
CREATE TABLE livre
(
bookid INT(10) NOT NULL,
bookname VARCHAR(255) NULL,
bookprice INT(3) NOT NULL,
booktype VARCHAR(10) NULL,
bookcenturypublished INT(2) NOT NULL
);
```

CREATE TABLE student

studentnationality VARCHAR(255) NULL,
